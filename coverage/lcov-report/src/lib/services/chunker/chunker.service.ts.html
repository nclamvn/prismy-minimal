export type ChunkingTier = 'basic' | 'standard' | 'premium'
 
export class ChunkerService {

    this.strategies = new Map()
    this.strategies.set('basic', new SimpleStrategy())

    this.strategies.set('premium', new SmartProductionStrategy())
